#include <iostream>
using namespace std;

int main()
{
  int alphabet[26] = {};
  int size;
  cin >> size;
  int count = 0;
  for (int i = 0; i < size; i++) {
    char a;
    cin >> a;
    alphabet[int(a) - 65]++;
  }
  for (int i = 0; i <= 26; i++) {
    if (alphabet[i] > 0)
      count++;
  }
  cout << count << endl;
  for (int i = 0; i <= 26; i++) {
    if (alphabet[i] > 0)
      cout << char(65 + i) << " " << alphabet[i] << endl;
  }
  return 0;
}

#include <iostream>
using namespace std;

int main()
{
  int numbers[1000];
  int size;
  cin >> size;
  for (int i = 0; i < size; i++) {
    int a;
    cin >> a;
    numbers[i] = a;
  }
  int temp;
  for (int i = 0; i < size-1; i++) {
    for (int j = i + 1; j < size; j++) {
      if (numbers[j] < numbers[i])
      {
        temp = numbers[i];
        numbers[i] = numbers[j];
        numbers[j] = temp;
      }
    }
  }
  for (int i = 0; i < size; i++)
    cout << numbers[i] << " ";
  
  return 0;
}
