  }
  for (int i = 0; i <= 26; i++) {
    if (alphabet[i] > 0)
      count++;
  }
  cout << count << endl;

    int a;

    numbers[i] = a;
  }
  int temp;
  for (int i = 0; i < size-1; i++) {
    for (int j = i + 1; j < size; j++) {
      if (numbers[j] < numbers[i])
      {
        temp = numbers[i];
        numbers[i] = numbers[j];
        numbers[j] = temp;
      }
    }
  }
  for (int i = 0; i < size; i++)

  return 0;
}

int main()
{
	int numbers[100][3];
	int size;
	cin >> size;
	for (int i = 0; i < size; i++) {
		for (int j = 0; j < 3; j++) 
		cin >> numbers[i][j];
	}
	int tempH,tempM,tempS;
	for (int i = 0; i < size - 1; i++) {
		for (int j = i + 1; j < size; j++) {
			int totalTimePrev = numbers[i][0] * 3600 + numbers[i][1] * 60 + numbers[i][2];
			int totalTimeCurr = numbers[j][0] * 3600 + numbers[j][1] * 60 + numbers[j][2];
			if (totalTimePrev > totalTimeCurr)
			{
				tempH = numbers[i][0];
				tempM = numbers[i][1];
				tempS = numbers[i][2];
				numbers[i][0] = numbers[j][0];
				numbers[i][1] = numbers[j][1];
				numbers[i][2] = numbers[j][2];
				numbers[j][0] = tempH;
				numbers[j][1] = tempM;
				numbers[j][2] = tempS;
			}
		}
	}
	for (int i = 0; i < size; i++)
			cout << numbers[i][0] << " " << numbers[i][1]<< " "<<numbers[i][2]<<endl;

	return 0;

{

	bool isPrime = true;
	for (int i = 2; i < k; i++) {

				isPrime = false;

			}
